cmake_minimum_required(VERSION 2.8)
set(PROJECT_NAME example)
project(${PRPOJECT_NAME})
set(Boost_USE_STATIC_LIBS ON)
find_package(Boost 1.56 COMPONENTS log log_setup thread filesystem system regex date_time chrono REQUIRED)
set(CMAKE_INCLUDE_CURRENT_DIR ON)
include_directories(SYSTEM ${Boost_INCLUDE_DIRS})
add_executable(${PRPOJECT_NAME} example.cpp)

if(WIN32)
  target_link_libraries(${PRPOJECT_NAME} ${Boost_LIBRARIES})
else(WIN32)
  add_definitions(-Wall -g -Os)
  set_target_properties(${PRPOJECT_NAME} PROPERTIES COMPILE_FLAGS "-std=c++11")
  target_link_libraries(${PRPOJECT_NAME} ${Boost_LIBRARIES} pthread rt)
endif(WIN32)
